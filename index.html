<script>

    // 测试题目数据 [cite: 149-350]

    const questions = [

        { id: 1, text: "周末早晨醒来，你第一件事想：", options: [

            { text: "今天约谁一起玩呢？", value: "A" },

            { text: "终于有整块时间做自己的事了", value: "B" },

            { text: "今天要完成哪些计划？", value: "C" },

            { text: "看看有什么新鲜事会发生", value: "D" }

        ]},

        // ... 此处省略中间题目，逻辑保持不变 ...

        { id: 2, text: "在派对上，你通常是：", options: [

            { text: "主动认识新朋友，气氛组担当", value: "A" },

            { text: "和熟悉的人聊天，或找个角落观察", value: "B" },

            { text: "有目的地交流，顺便拓展人脉", value: "C" },

            { text: "灵感迸发，成为话题中心", value: "D" }

        ]}

        // 注意：实际应用中请补充完整文档中的20道题

    ];

 

    // 性格描述与建议 [cite: 352-424]

    const typeDescriptions = {

        "社交型": { badge: "社交型", description: "你热情、共情、善于沟通 [cite: 355]", careers: ["HR/招聘", "公关/销售"], advice: "发挥人际敏感度，但注意设立边界 [cite: 362]" },

        "专业型": { badge: "专业型", description: "你专注、细致、逻辑性强 [cite: 366]", careers: ["数据分析师", "科研人员"], advice: "避免陷入过度分析 [cite: 373]" },

        "管理型": { badge: "管理型", description: "你条理清晰、执行力强 [cite: 377]", careers: ["项目经理", "运营管理"], advice: "增加灵活性训练 [cite: 384]" },

        "创意型": { badge: "创意型", description: "你想象力丰富、视角独特 [cite: 388]", careers: ["设计师", "产品经理"], advice: "学会适应必要的规则 [cite: 395]" }

    };

 

    let answers = {};

 

    // 渲染题目 [cite: 433-456]

    function renderQuiz() {

        const quizContainer = document.getElementById('quiz');

        quizContainer.innerHTML = questions.map(q => `

            <div class="question" id="question-${q.id}">

                <div class="question-title">第${q.id}题：${q.text}</div>

                <div class="options">

                    ${q.options.map(opt => `<div class="option" onclick="selectOption(${q.id}, '${opt.value}', event)">${opt.text}</div>`).join('')}

                </div>

            </div>

        `).join('');

    }

 

    // 选择选项逻辑 [cite: 458-473]

    function selectOption(qId, val, event) {

        answers[qId] = val;

        const parent = document.getElementById(`question-${qId}`);

        parent.querySelectorAll('.option').forEach(opt => opt.classList.remove('selected'));

        event.target.classList.add('selected');

       

        document.getElementById('submitBtn').disabled = Object.keys(answers).length < questions.length;

        updateProgress();

    }

 

    function updateProgress() {

        const progress = (Object.keys(answers).length / questions.length) * 100;

        document.getElementById('progress').style.width = `${progress}%`;

    }

 

    // 显示结果 [cite: 481-544]

    function showResult() {

        const counts = { A: 0, B: 0, C: 0, D: 0 };

        Object.values(answers).forEach(a => counts[a]++);

 

        let primaryType = "社交型";

        if (counts.B > counts.A) primaryType = "专业型";

        if (counts.C > Math.max(counts.A, counts.B)) primaryType = "管理型";

        if (counts.D > Math.max(counts.A, counts.B, counts.C)) primaryType = "创意型";

 

        const info = typeDescriptions[primaryType];

        document.getElementById('quiz').style.display = 'none';

        document.getElementById('submitBtn').style.display = 'none';

        document.getElementById('result').style.display = 'block';

        document.getElementById('resultContent').innerHTML = `

            <div style="text-align:center"><span class="type-badge">${info.badge}</span></div>

            <p><strong>特质：</strong>${info.description}</p>

            <p><strong>职业建议：</strong>${info.careers.join('、')}</p>

            <p><strong>建议：</strong>${info.advice}</p>

        `;

    }

 

    function restartQuiz() {

        location.reload(); // 简易重置方式

    }

 

    document.getElementById('submitBtn').addEventListener('click', showResult);

    window.onload = renderQuiz;

</script>
